\documentclass{article}
\usepackage[paperheight=7cm,paperwidth=16cm,margin=0cm]{geometry}
\usepackage{amsmath}
\usepackage{tikz}

\begin{document}
	
% Dimensions
\def\xmin{-0.9}		% x-axis minimum
\def\xmax{1.0}		% x-axis maximum
\def\a{0.5}         % Aperture radius
\def\za{0.06}       % Aperture thickness
\def\x1{0.4}		% source coordinate

\def\zmin{-0.6}		% z-axis minimum
\def\zmax{4.2}		% z-axis maximum

\def\r{4.0}			% r-vector length
\def\zr{3.5}		% Rayleih distance

\def\angle{12}		% r-vector angle (theta)

\def\ztext{0.6}		% z-position for text
	
% Colours
\colorlet{aperturecolor}{red!80!black}		% Aperture, vibrating surface
\colorlet{pathcolor}{blue!90!black}			% Sound paths (vectors)

% Styles  
\tikzstyle{axisline}=[black, ->, thick] 
\tikzstyle{Rline} = [pathcolor, ->, thick]
\tikzstyle{rline} = [pathcolor, ->, thick, dotted]
\tikzstyle{wall} = [color= blue!60!white, opacity=0.1]

\tikzstyle{information text}=[rounded corners,fill=white, inner sep=1ex]

\begin{tikzpicture}[scale=3]%[>=stealth]

% Wall
\fill[wall] (\zmin, \xmin) rectangle  (0.0, \xmax);
\fill[fill=aperturecolor, opacity=0.5] (-\za, -\a) rectangle (0.0, \a);

% Axes
\draw[axisline] (\zmin, 0) -- (\zmax, 0)  node(zaxis)[below]{$z$};
\draw[axisline] (0, \xmin) -- (0, \xmax)  node(xaxis)[right]{$x$};

\draw[color=aperturecolor, thick, |-|] (-0.15, -\a) -- (-0.15, \a)
	node[near end, anchor=south east] {$d=2a$};

% Sound paths
\draw[Rline] (0, \a) -- (\zr,0)
	node[near start, anchor=south west]{$\vec{R}$};

\node[anchor=north] at (\zr, 0) {$z_R$};


% Annotations
\draw[aperturecolor](0, -\x1) -- ++(0.10, -0.05)
	node[aperturecolor, anchor=west] {$S_1$};

\node [anchor=north west ,text width=12cm,style=information text] at (0, \xmin)
	{ Rayleigh distance $z_R$ defined by $R_{max} = z_R +\tfrac{1}{4} \lambda $ gives 
			$z_R = \frac{d^2}{2\lambda} + \tfrac{1}{8} \lambda \approx \frac{d^2}{2\lambda}$ 
	};

\end{tikzpicture}

\end{document}
